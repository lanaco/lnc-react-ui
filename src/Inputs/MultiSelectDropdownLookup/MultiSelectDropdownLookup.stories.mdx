import { Meta, Story, Canvas, Props } from "@storybook/addon-docs/blocks";
import { ArgsTable, Stories } from "@storybook/addon-docs";
import MultiSelectDropdownLookup from "./index";

<Meta title="Inputs/Multi Select Dropdown Lookup" component={MultiSelectDropdownLookup} />

export const apiData = [
  { value: "1", label: "Birch" },
  { value: "2", label: "Oak" },
  { value: "3", label: "Maple" },
  { value: "4", label: "Pine" },
];

export const TemplateDefault = (args) => {
  const filterOptions = (inputValue) => {
    return apiData.filter((i) =>
      i.label.toLowerCase().includes(inputValue.toLowerCase())
    );
  };
  const loadOptions = (inputValue, callback) => {
    setTimeout(() => {
      callback(filterOptions(inputValue));
    }, 1000);
  };
  return (
    <div style={{ height: "130px" }}>
      <MultiSelectDropdownLookup
        loadOptions={loadOptions}
        defaultOptions={true}
        {...args}
      />
    </div>
  );
};
