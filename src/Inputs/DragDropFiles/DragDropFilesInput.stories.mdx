import { Meta, Story, Canvas, Props } from "@storybook/addon-docs/blocks";
import { ArgsTable, Stories } from "@storybook/addon-docs";
import DragDropFiles from "./DragDropFiles";
import DragAndDropFile from "../DragAndDropFile/DragAndDropFile";
import UploadedFile from "../../General/UploadedFile/UploadedFile";
import Chip from "../../Data display/Chip/Chip";
import { Prism as SyntaxHighlighter } from "react-syntax-highlighter";
import { nord } from "react-syntax-highlighter/dist/esm/styles/prism";

<Meta title="Inputs/Drag & Drop Files" component={DragDropFiles} />

export const TemplateDefault = (args) => <DragDropFiles {...args} />;

export const TemplateAcceptImages = (args) => (
  <DragDropFiles
    multiple={true}
    acceptDropzone={{
      "image/jpeg": [],
      "image/jpg": [],
      "image/png": [".png"],
    }}
    acceptInput={"image/*"}
    {...args}
  />
);

export const TemplateWithFiles = (args) => (
  <DragDropFiles
    showFileSize={true}
    files={[
      { name: "File 1", size: 20000000 },
      { name: "File 2", size: 1500 },
    ]}
    {...args}
  />
);

export const TemplateDragAndDropFile = (args) => <DragAndDropFile {...args} />;

export const TemplateCustomControl = (args) => (
  <DragAndDropFile control={<Chip>Custom control</Chip>} {...args} />
);

export const TemplateDisabled = (args) => (
  <DragAndDropFile
    disabled={true}
    selectFileText={"FILE"}
    dndFileText={"This is the place where you should drag and drop your file"}
    {...args}
  />
);

export const TemplateSizes = (args) => (
  <div>
    <DragDropFiles {...args} size="small" />
    <DragDropFiles {...args} size="medium" />
    <DragDropFiles {...args} size="large" />
  </div>
);

export const TemplateColors = (args) => (
  <div>
    <DragDropFiles {...args} color="primary" />
    <DragDropFiles {...args} color="secondary" />
    <DragDropFiles {...args} color="success" />
    <DragDropFiles {...args} color="warning" />
    <DragDropFiles {...args} color="danger" />
    <DragDropFiles {...args} color="information" />
    <DragDropFiles {...args} color="neutral" />
  </div>
);
