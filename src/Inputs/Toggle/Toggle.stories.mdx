<Meta title="Inputs/Toggle" component={Toggle} />

export const TemplateDefault = (props) => {
  const [checked, setChecked] = useState(props.value || false);
  return (
    <Toggle {...props} value={checked} onChange={(value) => setChecked(!checked)} />
  );
};

export const TemplateTrueFalse = (args) => (
  <div>
    <Toggle {...args} />
    <Toggle value={true} {...args} />
  </div>
);

export const TemplateLabel = (args) => (
  <div style={{ width: "220px" }}>

    <Toggle {...args} label={"Label as string"} labelPosition="left" />
    <Toggle
      {...args}
      label={<span>Label as component</span>}
      value={true}
      labelPosition="left"
    />
    <Toggle {...args} label={"Label as string"} labelPosition="right" spaceBetween={true} />
    <Toggle
      {...args}
      label={<span>Label as component</span>}
      value={true}
      labelPosition="left"
      spaceBetween={true}
    />
  </div>
);
