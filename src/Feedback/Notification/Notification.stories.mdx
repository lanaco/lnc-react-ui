export const TemplateDefault = (args) => {
  const notify = () => notification("Message");

    <div>

};

export const TemplateCustom= (args) => {
  const notify = () => notification(<NotificationMessage icon="user" title={"Title"}>Message</NotificationMessage>);
  return (
    <div>
      <button onClick={() => notification(<NotificationMessage icon="user" title={"Title"}>Message</NotificationMessage>)}>With Icon</button>
      <br /><br />
      <button onClick={() => notification(<NotificationMessage avatar={true} avatarProps={{ image: logo }} title={"Title"} actions={<><button>Confirm</button><button>Dismiss</button></>}>Message</NotificationMessage>)}>With Action</button>
      <br /><br />
      <button onClick={() => notification(<NotificationMessage avatar={true} inlineActions={true} avatarProps={{ image: logo }} title={"Title"} actions={<a>Dismiss</a>}>Message</NotificationMessage>)}>With Inline Actions</button>
      <br /><br />
      <button onClick={() => notification(<NotificationMessage sideButtons={true} avatar={true} avatarProps={{ image: logo }}  title={"Title"} actions={<><button>Confirm</button><button>Dismiss</button></>}>Message</NotificationMessage>, {closeButton: false})}>With Side Buttons</button>
      <NotificationContainer {...args}/>
    </div>
  );
};
