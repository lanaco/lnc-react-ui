export const TemplateCustom = (args) => {
  const [open, setOpen] = useState(true);
  return (
    <>
      {open && (
        <Alert
          open={open}
          color={"secondary"}
          statusIcon={"user-circle"}
          title="Custom alert"
          actions={
            <>
              <a onClick={() => alert("Connection failed")}>Connect</a>
              <a onClick={() => setOpen(false)}>Dismiss</a>
            </>
          }
          {...args}
        >
          Lorem ipsum dolor sit amet, consectetur adipiscing elit, sed do
          eiusmod tempor incididunt ut labore et sed.
        </Alert>
      )}
      <br />
      <Button disabled={open} onClick={() => setOpen(!open)}>
        Open
      </Button>
    </>
  );
};
