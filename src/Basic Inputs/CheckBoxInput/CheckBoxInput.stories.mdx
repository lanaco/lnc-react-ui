import { Meta, Story, Canvas, Props } from "@storybook/addon-docs/blocks";
import { ArgsTable, Stories } from "@storybook/addon-docs";
import CheckBoxInput from "./index";
import Icon from "../../General/Icon";
import Button from "../../General/Button";
import image from "../../_utils/images/testsvg.svg";
import { useRef, useState } from "react";

<Meta title="Basic Inputs/Check Box Input" component={CheckBoxInput} />

export const TemplateDefault = (args) => <CheckBoxInput {...args} />;

export const TemplateIndeterminate = (args) => {
  const checkboxRef = useRef();
  return <>
  <CheckBoxInput ref={checkboxRef} {...args} indeterminate={true} label="Indeterminate" />
  <Button onClick={(e) => { let checkbox = checkboxRef.current; checkbox.indeterminate = true; }}>Set Indeterminate</Button>
  </>
};

export const TemplateDisabled = (args) => (
  <>
    <CheckBoxInput
      {...args}
      disabled={true}
      label="Disabled"
      labelPosition="left"
    />
    <CheckBoxInput
      {...args}
      disabled={true}
      checked={true}
      label="Disabled"
      labelPosition="left"
    />
  </>
);

export const TemplateReadOnly = (args) => (
  <div style={{ display: "flex", gap: "12px" }}>
    <CheckBoxInput {...args} readOnly={true} defaultChecked={true} />
    <CheckBoxInput {...args} readOnly={true} />
  </div>
);

export const TemplateCustom = (args) => {
 const [checked, setChecked] = useState(true);
 return <CheckBoxInput
    customCheckmark={<img src={image} alt="" />}
    label="Custom"
    checked={checked}
    onChange={(e) => setChecked(!checked)}
  ></CheckBoxInput>
};

export const TemplateSpaceBetween = (args) => <>
<CheckBoxInput label="Label" spaceBetween={true} defaultChecked={true} {...args} />
<hr />
<CheckBoxInput label="Label" labelPosition="left" spaceBetween={true} defaultChecked={true} {...args} />
</>;


export const TemplateSizes = (args) => (
  <div style={{ display: "flex", gap: "12px" }}>
    <CheckBoxInput {...args} size="small" />
    <CheckBoxInput {...args} size="medium" />
    <CheckBoxInput {...args} size="large" />
  </div>
);

export const TemplateColors = (args) => (
  <div style={{ display: "flex", gap: "12px" }}>
    <CheckBoxInput {...args} color="primary" />
    <CheckBoxInput {...args} color="secondary" />
    <CheckBoxInput {...args} color="success" />
    <CheckBoxInput {...args} color="warning" />
    <CheckBoxInput {...args} color="danger" />
    <CheckBoxInput {...args} color="information" />
  </div>
);
